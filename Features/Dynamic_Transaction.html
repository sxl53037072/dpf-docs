动态标签

  • 下面以这些事务方法来了解配置文件各个动态标签的含义
  1. include及sql标签
    • include: 引入外部的sql脚本，以refid指定引入的id.
    • sql: 预定义sql片段，可以在当前事务文件中共享使用
         <transactions>
        <sql id="pageEndSql">
            <![CDATA[
              ) tz ) ttz
              WHERE
              ttz.row_num <= :endNum
              and
              ttz.row_num >= :beginNum
            ]]>
        </sql>
        <sql id="pageBeginSql">
            SELECT
            ttz.*
            FROM
            (SELECT CAST(@rownum :=@rownum + 1 as SIGNED) AS row_num, tz.*
            FROM(SELECT @rownum := 0) r,(
        </sql>
        <sql id="whereSql">
            and NAME like CONCAT(CONCAT('%',:name),'%')
            and age = :age
            <if test="n != null">
                and 1 = 2
            </if>
        </sql>
        <!--查询-->
        <transaction name="SelectBaseTest.tran.select" event="select" return="true">
            <sql>
                <include refid="pageBeginSql" />
                  select * from
                    t_test_tran t
                  where 1 = 1
                <include refid="whereSql" />
                <include refid="pageEndSql" />
            </sql>
        </transaction>
     </transactions>
  2. where及if标签
    • where: where包含的内容如果以and或者or开头剔除这些关键字,如果包含内容为空则删除where。
    • if: 根据if的条件判断是否需要引入if元素中的内容，可以多重嵌套
    <!--多层if查询-->
    <transaction name="SelectBaseTest.tran.if.more.select" event="select" return="true">
        <sql>

            select * from
            t_test_tran t
            <where>
                <if test="arg1!= null">
                    and 1 = 1
                    <if test="arg3 !=null">
                        and 1 = 3
                        <if test="arg6  != null">
                            and 1 = 6
                        </if>
                        <if test="arg7 !=  null">
                            and 1 = 7
                        </if>
                        <if test="arg8!=null">
                            and 1 = 8
                            <if test="arg9 != null ">
                                and 1 = 9
                            </if>
                            <if test=" arg10 != null">
                                and 1 = 10
                            </if>
                        </if>
                    </if>
                    <if test="arg4 != null">
                        and 1 = 4
                    </if>
                    <if test="arg5 != null">
                        and 1 = 5
                    </if>
                </if>
                <if test="arg2 != null">
                    and 1 = 2
                </if>
            </where>
        </sql>
    </transaction>
  3. choose, when, otherwise
    • choose: 条件判断语句用来包含when和otherwise标签。
    • when: 根据条件判断是否需要引入when元素中的内容，有且只能有一个when成立
    • otherwise: when条件全部未成立则引入otherwise内容
      <!--choose查询-->
  <transaction name="SelectBaseTest.tran.choose.select" event="select" return="true">
      <sql>

          select * from
          t_test_tran t
          <where>
              <choose>
                  <when test="arg2 !=null ">
                      id = :arg2
                  </when >
                  <when test="arg3 !=null ">
                      id = :arg3
                  </when >
                  <when test=" arg4 != null ">
                      id = :arg4
                  </when >
                  <otherwise>
                      1 = 1
                  </otherwise>
              </choose>
          </where>
      </sql>
  </transaction>
  4. foreach
    • foreach: 循环指定的list数组生成sql语句，用到sql注入标签${},需要慎重防止sql注入。
      • collection: 循环数组名
      • index: 循环索引从0开始
      • item: 每次循环的对象，在主体内容中只能使用${}标签进行引入变量
      • open: 开始循环前缀
      • separator: 每次循环分隔符
      • close: 结束循环后缀
       <transactions>     
     <transaction name="SelectBaseTest.tran.foreach.select" event="select" return="true">
         <sql>
             select * from
             t_test_tran t
             <where>
                 id in
                 <if test="list != null">
                     <foreach collection="list" index="index" item="item" open="(" separator="," close=")">
                            ${item}
                     </foreach>
                 </if>

             </where>
         </sql>
     </transaction>

     <!--foreach删除-->
       <transaction name="DeleteBaseTest.tran.foreach.delete" event="delete">
           <sql>
               delete from t_test_tran where id in
               <foreach collection="list" index="index" item="item" open="(" separator="," close=")">
                   ${item.id}
               </foreach>
           </sql>
       </transaction>
  </transactions>
  5. bind
    • bind: 绑定参数，或者对现有参数进行修改，可以直接引入java类中静态变量，具体参考ognl表达式
      • name: 参数名，用来在sql中进行引用
      • value: 参数值，直接使用ognl表达式
    <transactions>
    <transaction name="SelectBaseTest.tran.bind.select" event="select" return="true">
        <sql>
            select * from
            t_test_tran t
            <where>
                <if test="arg2 != null">
                    <bind name="arg" value="'2000000'+arg2" />
                    id in (${arg})
                </if>
                <if test="arg3 != null">
                    <bind name="arg" value="'2000000'+arg3" />
                    id = :arg
                </if>
                <if test="arg4 != null">
                    <bind name="arg" value="'2000000'+arg4" />
                    id = ${arg}
                </if>
            </where>
        </sql>
    </transaction>

    <!--静态数据引入 查询-->
    <transaction name="SelectBaseTest.tran.static.select" event="select" return="true">
        <sql>
            select * from t_test_tran
            <bind name="id" value="@integration.select.SelectBaseTest@ID" />
            where id = ${id}
        </sql>
    </transaction> 
 </transactions>
  6. trim
    • trim: trim 是更灵活用来去除多余关键字的标签，where set标签基于trim创建
      • prefix: 前缀覆盖并增加其内容
      • prefixOverrides: 前缀判断的条件
      • suffix: 后缀覆盖并增加其内容
      • suffixOverrides:后缀判断的条件
    <transactions>
  <!--trim 查询-->
  <transaction name="SelectBaseTest.tran.trim.select" event="select" return="true">
      <sql>
          select * from t_test_tran
          <trim prefix="where" prefixOverrides="AND|OR">
              <if test="id!=null">
                  and id=:id
              </if>
              <if test="name!=null">
                  and name=:name
              </if>
          </trim>
      </sql>
  </transaction>   
  <!--limit 查询-->
  <transaction name="SelectBaseTest.tran.limit.select" event="select" return="true">
      <sql>
          select * from t_test_tran
          <trim prefix="limit" prefixOverrides=",">
              <if test="begin != null">
                  ${begin}
              </if>
              <if test="end != null">
                  ,${end}
              </if>
          </trim>
      </sql>
  </transaction>
</transactions>
  7. before,after
    • before: 事务前置方法，基于ognl表达式，方法首个参数为map类型，map.get("params")可以用于获取事务参数(类型为map),在方法中对该参数修改后返回可以改变参数值
    • after: 事务后置方法，基于ognl表达式，方法首个参数为map类型，map.get("result")可以获取事务返回数组，可修改后返回
         <!--静态前后置数据引入-->
     <transaction name="SelectBaseTest.tran.static.around.select" event="select" return="true"
                  before="@integration.select.SelectBaseTest@before"
                  after="@integration.select.SelectBaseTest@after">
         <sql>
             select * from t_test_tran
             where id = :a
         </sql>
     </transaction>
    java静态方法 
         public class SelectBaseTest{
         public static Object before(Map map){
             Map params = (Map) map.get("params");
             //获取事务bean
             TransactionService transactionService = (TransactionService) BeanLoader.getBean("transactionService");
             String a = transactionService.doTranString("SelectBaseTest.tran.static.before.id.select",null);
             params.put("a", a);
             return params;
         }
         public static Object after(Map map){
             List result = (ArrayList) map.get("result");
             result.add(new HashMap(){{
                 put("id", "4");
                 put("name", "4");
                 put("age", "4");
                 put("email", "4");
                 put("phone", "4");
                 put("remark", "4");
                 put("create_time", "2018-09-14 15:07:43.0");
             }});
             return result;
         }
      }
  8. set
    • set: set内容如果以逗号结尾则删除逗号，如果set中无内容则删除set标签
         <!--更新set-->
         <transaction name="UpdateBaseTest.tran.set.update" event="insert">
             <sql>
                 update t_test_tran
                 <set>
                     <if test="name != null">
                         name=:name,
                     </if>
                     <if test="age != null">
                         age=:age,
                     </if>
                     <if test="email != null">
                         email=:email,
                     </if>
                 </set>
                 where id = :id
             </sql>
         </transaction>
